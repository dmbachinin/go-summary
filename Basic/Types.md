```go
    // low - –ò–Ω–¥–µ–∫—Å, —Å –∫–æ—Ç–æ—Ä–æ–≥–æ –Ω–∞—á–∏–Ω–∞–µ—Ç—Å—è –Ω–æ–≤—ã–π —Å—Ä–µ–∑ (–≤–∫–ª—é—á–∏—Ç–µ–ª—å–Ω–æ). –ï—Å–ª–∏ –æ–ø—É—â–µ–Ω, —Å—á–∏—Ç–∞–µ—Ç—Å—è 0
    // high - –ò–Ω–¥–µ–∫—Å, –¥–æ –∫–æ—Ç–æ—Ä–æ–≥–æ –∏–¥—ë—Ç –Ω–æ–≤—ã–π —Å—Ä–µ–∑ (–Ω–µ –≤–∫–ª—é—á–∞—è high). –ï—Å–ª–∏ –æ–ø—É—â–µ–Ω, —Å—á–∏—Ç–∞–µ—Ç—Å—è len(s). 
    //  * –û–ø—Ä–µ–¥–µ–ª—è–µ—Ç –¥–ª–∏–Ω—É –Ω–æ–≤–æ–≥–æ —Å—Ä–µ–∑–∞: len = high - low
    // max (—Ç–æ–ª—å–∫–æ –≤ —Ç—Ä—ë—Ö–ø–∞—Ä–∞–º–µ—Ç—Ä–∏—á–Ω–æ–π —Ñ–æ—Ä–º–µ) - –ò—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è –¥–ª—è –æ–≥—Ä–∞–Ω–∏—á–µ–Ω–∏—è –≤–æ–∑–º–æ–∂–Ω–æ—Å—Ç–∏ —Ä–æ—Å—Ç–∞ —Å—Ä–µ–∑–∞ —á–µ—Ä–µ–∑ append. –ï—Å–ª–∏ –æ–ø—É—â–µ–Ω,  —Ç–æ —Å—á–∏—Ç–∞–µ—Ç—Å—è cap(s). 
    //  * –û–ø—Ä–µ–¥–µ–ª—è–µ—Ç —ë–º–∫–æ—Å—Ç—å –Ω–æ–≤–æ–≥–æ —Å—Ä–µ–∑–∞: cap = max - low


    arr := [5]int{10, 20, 30, 40, 50}
    s := arr[:] // –æ—Ç –Ω–∞—á–∞–ª–∞ –¥–æ –∫–æ–Ω—Ü–∞ ‚Äî len=5, cap=5
    fmt.Println(s, len(s), cap(s)) // [10 20 30 40 50] 5 5

    // –î–≤—É—Ö–ø–∞—Ä–∞–º–µ—Ç—Ä–∏—á–Ω–∞—è —Ñ–æ—Ä–º–∞: [low:high]
    sub1 := s[1:4] // —Å –∏–Ω–¥–µ–∫—Å–∞ 1 –ø–æ 3
    fmt.Println(sub1, len(sub1), cap(sub1)) // [20 30 40] 3 4
    // len = high - low = 4 - 1 = 3
    // cap = cap(s) - low = 5 - 1 = 4

    // –ï—Å–ª–∏ –æ–ø—É—Å—Ç–∏—Ç—å low ‚Äî –±–µ—Ä—ë–º –æ—Ç –Ω–∞—á–∞–ª–∞
    sub2 := s[:3]
    fmt.Println(sub2, len(sub2), cap(sub2)) // [10 20 30] 3 5

    // –ï—Å–ª–∏ –æ–ø—É—Å—Ç–∏—Ç—å high ‚Äî –±–µ—Ä—ë–º –¥–æ –∫–æ–Ω—Ü–∞
    sub3 := s[2:]
    fmt.Println(sub3, len(sub3), cap(sub3)) // [30 40 50] 3 3

    // –¢—Ä—ë—Ö–ø–∞—Ä–∞–º–µ—Ç—Ä–∏—á–Ω–∞—è —Ñ–æ—Ä–º–∞: [low:high:max]
    sub4 := s[1:3:10] // len = 3 - 1 = 2, cap = max - low = 10 - 1 = 9
    fmt.Println(sub4, len(sub4), cap(sub4)) // [20 30] 2 9

    // –ß–µ—Ä–µ–∑ make([]—Ç–∏–ø, –¥–ª–∏–Ω–∞, –µ–º–∫–æ—Å—Ç—å)
    //  * –¥–ª–∏–Ω–∞ (len) ‚Äî –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —ç–ª–µ–º–µ–Ω—Ç–æ–≤, –∫–æ—Ç–æ—Ä—ã–µ —Å—Ä–∞–∑—É –¥–æ—Å—Ç—É–ø–Ω—ã –¥–ª—è –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è
    //  * –µ–º–∫–æ—Å—Ç—å (cap) ‚Äî –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —ç–ª–µ–º–µ–Ω—Ç–æ–≤, –∫–æ—Ç–æ—Ä—ã–µ –º–æ–∂–Ω–æ –¥–æ–±–∞–≤–∏—Ç—å –±–µ–∑ –≤—ã–¥–µ–ª–µ–Ω–∏—è –Ω–æ–≤–æ–π –ø–∞–º—è—Ç–∏
    slice := make([]int, 4) // –¢–æ –∂–µ —Å–∞–º–æ–µ, —á—Ç–æ make([]int, 4, 4)
    slice := make([]int, 2, 4)
    // –ß–µ—Ä–µ–∑ var (–¥–ª–∏–Ω–Ω–∞—è —Ñ–æ—Ä–º–∞)
    var slice2 []int = make([]int, 2, 5)

    fmt.Println(slice)      // [0 0]
    fmt.Println(len(slice)) // 2
    fmt.Println(cap(slice)) // 4
```

#### append

`append` ‚Äî –≤—Å—Ç—Ä–æ–µ–Ω–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è, –∫–æ—Ç–æ—Ä–∞—è **–í–°–ï–ì–î–ê** –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –Ω–æ–≤—ã–π —Å—Ä–µ–∑, –≤ –∫–æ—Ç–æ—Ä—ã–π –¥–æ–±–∞–≤–ª–µ–Ω—ã —ç–ª–µ–º–µ–Ω—Ç—ã –∫ –∏—Å—Ö–æ–¥–Ω–æ–º—É —Å—Ä–µ–∑—É.
–í–∞–∂–Ω—ã–π –º–æ–º–µ–Ω—Ç: append –º–æ–∂–µ—Ç –≤–µ—Ä–Ω—É—Ç—å —Å—Ä–µ–∑, —É–∫–∞–∑—ã–≤–∞—é—â–∏–π –Ω–∞ —Ç–æ—Ç –∂–µ –º–∞—Å—Å–∏–≤, –∞ –º–æ–∂–µ—Ç —Å–æ–∑–¥–∞—Ç—å –Ω–æ–≤—ã–π –º–∞—Å—Å–∏–≤ –∏ —Å–∫–æ–ø–∏—Ä–æ–≤–∞—Ç—å –¥–∞–Ω–Ω—ã–µ, –µ—Å–ª–∏ –≤ —Å—Ç–∞—Ä–æ–º –Ω–µ—Ç –º–µ—Å—Ç–∞.

–û—Å–æ–±–µ–Ω–Ω–æ—Å—Ç–∏:

- –†–∞–±–æ—Ç–∞–µ—Ç —Å nil-—Å—Ä–µ–∑–∞–º–∏ ‚Äî –ø—Ä–∏ –ø–µ—Ä–≤–æ–º append —Å–æ–∑–¥–∞—ë—Ç—Å—è –Ω–æ–≤—ã–π –º–∞—Å—Å–∏–≤.
- –ù–µ –∏–∑–º–µ–Ω—è–µ—Ç len –∏—Å—Ö–æ–¥–Ω–æ–≥–æ —Å—Ä–µ–∑–∞ –Ω–∞–ø—Ä—è–º—É—é ‚Äî –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –Ω–æ–≤—ã–π.
- –ü—Ä–∏ –Ω–µ—Ö–≤–∞—Ç–∫–µ cap —Å–æ–∑–¥–∞—ë—Ç –Ω–æ–≤—ã–π –º–∞—Å—Å–∏–≤ –∏ –∫–æ–ø–∏—Ä—É–µ—Ç –≤ –Ω–µ–≥–æ —Å—Ç–∞—Ä—ã–µ –¥–∞–Ω–Ω—ã–µ.
- –†–æ—Å—Ç —ë–º–∫–æ—Å—Ç–∏ (cap) –ø—Ä–∏ –ø–µ—Ä–µ–≤—ã–¥–µ–ª–µ–Ω–∏–∏ ‚Äî –≤–Ω—É—Ç—Ä–µ–Ω–Ω–∏–π –∞–ª–≥–æ—Ä–∏—Ç–º runtime:
- –û–±—ã—á–Ω–æ cap —É–¥–≤–∞–∏–≤–∞–µ—Ç—Å—è, –ø–æ–∫–∞ < 1024, –∑–∞—Ç–µ–º —Ä–∞—Å—Ç—ë—Ç –ø—Ä–∏–º–µ—Ä–Ω–æ –Ω–∞ 25%.
- –ï—Å–ª–∏ cap –ø–æ–∑–≤–æ–ª—è–µ—Ç, append –ø—Ä–æ—Å—Ç–æ –∑–∞–ø–∏—Å—ã–≤–∞–µ—Ç —ç–ª–µ–º–µ–Ω—Ç—ã –≤ —Ç–µ–∫—É—â–∏–π –º–∞—Å—Å–∏–≤(**–º–æ–≥—É—Ç –±—ã—Ç—å –≤–∏–¥–Ω—ã –≤ –¥—Ä—É–≥–∏—Ö —Å—Ä–µ–∑–∞—Ö**).

–ö–∞–∫ —Ä–∞–±–æ—Ç–∞–µ—Ç –ø–æ—à–∞–≥–æ–≤–æ:

1. –ü—Ä–æ–≤–µ—Ä–∫–∞ —ë–º–∫–æ—Å—Ç–∏
   - –ï—Å–ª–∏ len(s)+len(elems) <= cap(s), –º–µ—Å—Ç–∞ —Ö–≤–∞—Ç–∞–µ—Ç ‚Üí –∑–∞–ø–∏—Å—ã–≤–∞–µ–º –≤ —Å—É—â–µ—Å—Ç–≤—É—é—â–∏–π –º–∞—Å—Å–∏–≤.
   - –ï—Å–ª–∏ –º–µ—Å—Ç–∞ –Ω–µ—Ç ‚Üí –≤—ã–¥–µ–ª—è–µ–º –Ω–æ–≤—ã–π –º–∞—Å—Å–∏–≤ —Å —É–≤–µ–ª–∏—á–µ–Ω–Ω–æ–π —ë–º–∫–æ—Å—Ç—å—é, –∫–æ–ø–∏—Ä—É–µ–º –¥–∞–Ω–Ω—ã–µ, –¥–æ–±–∞–≤–ª—è–µ–º –Ω–æ–≤—ã–µ —ç–ª–µ–º–µ–Ω—Ç—ã.
2. –í–æ–∑–≤—Ä–∞—Ç –Ω–æ–≤–æ–≥–æ —Å—Ä–µ–∑–∞
   - –ù–æ–≤—ã–π —Å—Ä–µ–∑ —Å—Å—ã–ª–∞–µ—Ç—Å—è –Ω–∞ —Ç–æ—Ç –∂–µ –∏–ª–∏ –Ω–æ–≤—ã–π –º–∞—Å—Å–∏–≤.
   - **–ò—Å—Ö–æ–¥–Ω—ã–π —Å—Ä–µ–∑ (s) –æ—Å—Ç–∞—ë—Ç—Å—è –ø—Ä–µ–∂–Ω–∏–º**.

```go

    // –î–æ–±–∞–≤–ª–µ–Ω–∏–µ —ç–ª–µ–º–µ–Ω—Ç–æ–≤ append(slice, values...)
    s := make([]int, 2, 4) // s=[0 0] len=2 cap=4
    s2 := append(s, 10) // s2=[0 0 10] len=3 cap=4
    s3 := append(s2, 20, 30) // s3=[0 0 10 20 30] len=5 cap=8 (–º–µ—Å—Ç–∞ –Ω–µ —Ö–≤–∞—Ç–∞–µ—Ç ‚Üí –Ω–æ–≤—ã–π –º–∞—Å—Å–∏–≤)
    s2[0] = 99 // –ü—Ä–æ–≤–µ—Ä–∏–º, –æ–±—â–∏–π –ª–∏ –º–∞—Å—Å–∏–≤:
    fmt.Println("s:", s) // s: [99 0]
    fmt.Println("s2:", s2) // s2: [99 0 10]
    fmt.Println("s3:", s3) // s3: [0 0 10 20 30] ‚Äî –Ω–µ –∏–∑–º–µ–Ω–∏–ª—Å—è, –∑–Ω–∞—á–∏—Ç –º–∞—Å—Å–∏–≤ –¥—Ä—É–≥–æ–π

    acc := &s2[1]
    acc.val = 100
    fmt.Println(s1, s2) // [{0}] [{0} {100}]
    s1 = append(s1, account{val: 1000}) // –î–∞–Ω–Ω–æ–µ –¥–µ–π—Å—Ç–≤–∏–µ –ü–ï–†–ï–ó–ê–ü–ò–®–ï–¢ –∑–Ω–∞—á–µ–Ω–∏–µ, –∫–æ—Ç–æ—Ä–æ–µ –±—ã–ª–æ –≤ –º–∞—Å—Å–∏–≤–µ
    fmt.Println(s1, s2) // [{0} {1000}] [{0} {1000}]
    
    slice1 := []int{1, 2, 3}
    slice2 := []int{4, 5, 6}
    slice1 = append(slice1, slice2...) // –ò—Å–ø–æ–ª—å–∑—É–µ–º `...` –¥–ª—è —Ä–∞—Å–∫—Ä—ã—Ç–∏—è –≤—Ç–æ—Ä–æ–≥–æ —Å—Ä–µ–∑–∞
    fmt.Println(slice1) // [1 2 3 4 5 6]

    slice1 := []int{} // len = 0, cap = 0
    slice1 = append(slice1, 4, 5, 6) // –î–æ–±–∞–≤–ª—è–µ–º –º–Ω–æ–∂–µ—Å—Ç–≤–æ –∑–Ω–∞—á–µ–Ω–∏–π
    fmt.Println(slice1) // [4 5 6], len = 3, cap = 3
```

##### –ü–∞–∫–µ—Ç slices (Go 1.21+)

`–î–û–ö–£–ú–ï–ù–¢–ê–¶–ò–Ø`: <https://pkg.go.dev/slices>

–ü–∞–∫–µ—Ç "slices" —Å–æ–¥–µ—Ä–∂–∏—Ç –æ–±–æ–±—â—ë–Ω–Ω—ã–µ —Ñ—É–Ω–∫—Ü–∏–∏ –¥–ª—è []T. –î–ª—è —Å—Ä–∞–≤–Ω–µ–Ω–∏—è –∏ —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∏ –µ—Å—Ç—å –¥–≤–µ –≤–µ—Ä—Å–∏–∏: ¬´–µ—Å—Ç–µ—Å—Ç–≤–µ–Ω–Ω–∞—è¬ª (–¥–ª—è —É–ø–æ—Ä—è–¥–æ—á–∏–≤–∞–µ–º—ã—Ö —Ç–∏–ø–æ–≤) –∏ —Å —Ñ—É–Ω–∫—Ü–∏–µ–π-–∫–æ–º–ø–∞—Ä–∞—Ç–æ—Ä–æ–º (*Func)

```go

        "slices" // –Ω–∞–±–æ—Ä —É—Ç–∏–ª–∏—Ç –¥–ª—è —Å—Ä–µ–∑–æ–≤ (Go 1.21+)
        "cmp" // —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω–∞—è –∫–æ–º–ø–∞—Ä–∞—Ü–∏—è –∑–Ω–∞—á–µ–Ω–∏–π (Go 1.21+)
    ) 
    
    // –°–†–ê–í–ù–ï–ù–ò–ï/–ü–†–û–í–ï–†–ö–ò
    
    // slices.Equal(x, y) ‚Äî —ç–ª–µ–º–µ–Ω—Ç–Ω–æ–µ —Ä–∞–≤–µ–Ω—Å—Ç–≤–æ

    y := []int{1, 2, 3}
    z := []int{1, 2, 4}

    fmt.Println(slices.Equal(x, z)) // false

    // slices.EqualFunc(x, y, eq) ‚Äî —Ä–∞–≤–µ–Ω—Å—Ç–≤–æ –ø–æ –ø—Ä–µ–¥–∏–∫–∞—Ç—É

    a := []P{{1,2}, {3,4}}
    b := []P{{1,2}, {3,4}}
    fmt.Println(slices.EqualFunc(a, b, eq)) // true

    // slices.Compare(x, y) ‚Äî –ª–µ–∫—Å–∏–∫–æ–≥—Ä–∞—Ñ–∏—á–µ—Å–∫–æ–µ —Å—Ä–∞–≤–Ω–µ–Ω–∏–µ; –∏—Å–ø–æ–ª—å–∑—É–µ—Ç <, >
    fmt.Println(slices.Compare([]int{1,2}, []int{1,2})) // 0
    fmt.Println(slices.Compare([]int{1,2}, []int{1,3})) // <0
    
    // –ü–û–ò–°–ö/–ü–†–û–í–ï–†–ö–ê –°–û–î–ï–†–ñ–ò–ú–û–ì–û

    // slices.Index(s, v) ‚Äî –∏–Ω–¥–µ–∫—Å –ø–µ—Ä–≤–æ–≥–æ –≤—Ö–æ–∂–¥–µ–Ω–∏—è –∏–ª–∏ -1
    s := []string{"go", "java", "go"}
    fmt.Println(slices.Index(s, "go"))   // 0
    fmt.Println(slices.Index(s, "rust")) // -1

    // slices.IndexFunc(s, pred) ‚Äî –∏–Ω–¥–µ–∫—Å –ø–æ –ø—Ä–µ–¥–∏–∫–∞—Ç—É

    odd := func(x int) bool { return x%2 == 1 }
    fmt.Println(slices.IndexFunc(s, odd)) // 1

    // slices.Contains(s, v) / ContainsFunc(s, pred) ‚Äî —Å–æ–¥–µ—Ä–∂–∏—Ç –ª–∏

    fmt.Println(slices.Contains(s, 2)) // true
    fmt.Println(slices.ContainsFunc(s, func(x int) bool { return x > 5 })) // false

    // –ú–û–î–ò–§–ò–ö–ê–¶–ò–ò –° –ö–û–ü–ò–†–û–í–ê–ù–ò–ï–ú (–ù–ï –≤—Å—Ç—Ä–æ–µ–Ω–Ω—ã–µ append/copy)

    c := slices.Clone(s) // slices.Clone(s) ‚Äî –ø–æ–≤–µ—Ä—Ö–Ω–æ—Å—Ç–Ω–∞—è –∫–æ–ø–∏—è
    
    s := []int{1, 2, 3, 4}
    s = slices.Replace(s, 1, 3, 9, 9) // -> [1, 9, 9, 4]

    // slices.Replace(s, i, j, repl...) ‚Äî –∑–∞–º–µ–Ω–∏—Ç—å –ø–æ–¥–æ—Ç—Ä–µ–∑–æ–∫ [i:j)
    s := []int{1, 2, 3, 4}
    s = slices.Replace(s, 1, 3, 9, 9) // -> [1, 9, 9, 4]

    // slices.Insert(s, i, vals...) ‚Äî –≤—Å—Ç–∞–≤–∫–∞ –ø–µ—Ä–µ–¥ –∏–Ω–¥–µ–∫—Å–æ–º i
    s := []int{1, 4}
    s = slices.Insert(s, 1, 2, 3) // -> [1, 2, 3, 4]

    // slices.Delete(s, i, j) ‚Äî —É–¥–∞–ª–∏—Ç—å –ø–æ–¥–æ—Ç—Ä–µ–∑–æ–∫ [i:j)
    s := []int{1, 2, 3, 4, 5}
    s = slices.Delete(s, 1, 4) // —É–¥–∞–ª–∏—Ç 2,3,4 -> [1,5]
    
    // –†–ê–ë–û–¢–ê –° –†–ê–ó–ú–ï–†–û–ú/–Å–ú–ö–û–°–¢–¨–Æ
    
    // slices.Grow(s, n) ‚Äî –≥–∞—Ä–∞–Ω—Ç–∏—Ä—É–µ—Ç cap(s) >= len(s)+n; –º–æ–∂–µ—Ç –≤–µ—Ä–Ω—É—Ç—å –Ω–æ–≤—ã–π —Å—Ä–µ–∑
    s := []int{1, 2, 3}
    s = slices.Grow(s, 1000) // –ø–æ–ª–µ–∑–Ω–æ –ø–µ—Ä–µ–¥ –±–æ–ª—å—à–∏–º –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ–º append

    // slices.Clip(s) ‚Äî —É—Å–µ—á—å —ë–º–∫–æ—Å—Ç—å –¥–æ –¥–ª–∏–Ω—ã (cap = len)
    s := make([]int, 0, 100)
    s = append(s, 1, 2, 3)
    s = slices.Clip(s) // –æ—Å–≤–æ–±–æ–∂–¥–∞–µ—Ç –ª–∏—à–Ω—é—é —ë–º–∫–æ—Å—Ç—å
    fmt.Println(len(s), cap(s)) // cap == 3

    // –£–î–ê–õ–ï–ù–ò–ï –î–£–ë–õ–ò–ö–ê–¢–û–í
    
    // slices.Compact(s) ‚Äî —É–¥–∞–ª—è–µ—Ç –ø–æ–¥—Ä—è–¥ –∏–¥—É—â–∏–µ –¥—É–±–ª–∏–∫–∞—Ç—ã (–¥–ª—è —Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã—Ö ‚Äî —É–±–∏—Ä–∞–µ—Ç –ø–æ–≤—Ç–æ—Ä—ã)
    s := []int{1,1,2,2,2,3,3}
    s = slices.Compact(s) // -> [1,2,3]

    // slices.CompactFunc(s, eq) ‚Äî ¬´—Å—Ö–ª–æ–ø—ã–≤–∞–µ—Ç¬ª –ø–æ–¥—Ä—è–¥ —Ä–∞–≤–Ω—ã–µ –ø–æ –ø—Ä–µ–¥–∏–∫–∞—Ç—É
    type P struct{ X, Y int }
    eq := func(a, b P) bool { return a.X == b.X && a.Y == b.Y }
    s := []P{{1,2},{1,2},{2,2}}
    s = slices.CompactFunc(s, eq) // -> [{1,2},{2,2}]

    // –°–û–†–¢–ò–†–û–í–ö–ê –ò –ü–†–û–í–ï–†–ö–ê –ü–û–†–Ø–î–ö–ê

    // slices.Sort(s) ‚Äî —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ –ø–æ –≤–æ–∑—Ä–∞—Å—Ç–∞–Ω–∏—é (–¥–ª—è ¬´—Å—Ä–∞–≤–Ω–∏–º—ã—Ö¬ª —Ç–∏–ø–æ–≤)
    s := []int{3, 1, 2}
    slices.Sort(s) // -> [1,2,3]

    // slices.SortFunc(s, less) ‚Äî —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ –ø–æ —Ñ—É–Ω–∫—Ü–∏–∏
    type U struct{ Name string; Age int }
    less := func(a, b U) int {
        // –°–Ω–∞—á–∞–ª–∞ –ø–æ Age, –∑–∞—Ç–µ–º –ø–æ Name
        if c := cmp.Compare(a.Age, b.Age); c != 0 { return c }
        return cmp.Compare(a.Name, b.Name)
    }
    s := []U{{"Bob", 30}, {"Ann", 20}, {"Ann", 25}}
    slices.SortFunc(s, less)

    // slices.IsSorted(s) / IsSortedFunc(s, less) ‚Äî –ø—Ä–æ–≤–µ—Ä–∫–∞ –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω–æ—Å—Ç–∏
    s := []int{1, 2, 3}
    fmt.Println(slices.IsSorted(s)) // true

    // –ë–ò–ù–ê–†–ù–´–ô –ü–û–ò–°–ö (–¥–ª—è –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã—Ö —Å—Ä–µ–∑–æ–≤)

    // slices.BinarySearch(s, target) ‚Äî –∏–Ω–¥–µ–∫—Å –∏ –Ω–∞–π–¥–µ–Ω–æ/–Ω–µ—Ç
    s := []int{1, 3, 5, 7}
    i, found := slices.BinarySearch(s, 5) // i=2, found=true

    // slices.BinarySearchFunc(s, x, cmp) ‚Äî –±–∏–Ω–∞—Ä–Ω—ã–π –ø–æ–∏—Å–∫ —Å –∫–æ–º–ø–∞—Ä–∞—Ç–æ—Ä–æ–º
    type W struct{ V string }
    s := []W{{"a"},{"c"},{"d"}}
    i, found := slices.BinarySearchFunc(s, W{"c"}, func(a, b W) int {
        return cmp.Compare(a.V, b.V)
    })
```

### –ö–∞–∫ —Ä–∞–±–æ—Ç–∞–µ—Ç –≤—ã–¥–µ–ª–µ–Ω–∏–µ –ø–∞–º—è—Ç–∏ –¥–ª—è —Å—Ä–µ–∑–æ–≤?

–ö–æ–≥–¥–∞ –º—ã —Å–æ–∑–¥–∞–µ–º —Å—Ä–µ–∑ —á–µ—Ä–µ–∑ make(), Go –≤–Ω—É—Ç—Ä–∏ –≤—ã–¥–µ–ª—è–µ—Ç –º–∞—Å—Å–∏–≤ –Ω—É–∂–Ω–æ–≥–æ —Ä–∞–∑–º–µ—Ä–∞ –∏ —Å–æ–∑–¥–∞–µ—Ç —Å—Å—ã–ª–∫—É –Ω–∞ —ç—Ç–æ—Ç –º–∞—Å—Å–∏–≤:

```plaintext
    –ü—Ä–∏–º–µ—Ä —Ä–∞–±–æ—Ç—ã make([]int, 3, 5) –≤ –ø–∞–º—è—Ç–∏
    –ú–∞—Å—Å–∏–≤ –≤ –ø–∞–º—è—Ç–∏:  [ 0  0  0  -  - ]  (—Ä–∞–∑–º–µ—Ä 5, cap=5)
    –°—Ä–µ–∑ —É–∫–∞–∑—ã–≤–∞–µ—Ç –Ω–∞: [ 0  0  0 ]
```

–ï—Å–ª–∏ –¥–æ–±–∞–≤–∏—Ç—å –Ω–æ–≤—ã–π —ç–ª–µ–º–µ–Ω—Ç —á–µ—Ä–µ–∑ append(), –æ–Ω –ø–æ–ø–∞–¥–µ—Ç –≤ —Ä–µ–∑–µ—Ä–≤

**–ö–æ–≥–¥–∞ –≤—ã–¥–µ–ª—è–µ—Ç—Å—è –Ω–æ–≤—ã–π –º–∞—Å—Å–∏–≤?**

Go –≤—ã–¥–µ–ª—è–µ—Ç –Ω–æ–≤—ã–π –º–∞—Å—Å–∏–≤ —Ç–æ–ª—å–∫–æ –µ—Å–ª–∏ –¥–æ–±–∞–≤–ª–µ–Ω–Ω—ã–π —ç–ª–µ–º–µ–Ω—Ç –ø—Ä–µ–≤—ã—à–∞–µ—Ç —Ç–µ–∫—É—â—É—é cap(slice).

- –ö–æ–≥–¥–∞ cap(slice) > len(slice)
–ï—Å–ª–∏ –¥–ª–∏–Ω–∞ (len) –º–µ–Ω—å—à–µ –µ–º–∫–æ—Å—Ç–∏ (cap), –Ω–æ–≤—ã–π —ç–ª–µ–º–µ–Ω—Ç –ø—Ä–æ—Å—Ç–æ –¥–æ–±–∞–≤–ª—è–µ—Ç—Å—è –≤ —Å–≤–æ–±–æ–¥–Ω–æ–µ –º–µ—Å—Ç–æ. –í—ã–¥–µ–ª–µ–Ω–∏—è –ø–∞–º—è—Ç–∏ –Ω–µ –ø—Ä–æ–∏—Å—Ö–æ–¥–∏—Ç.

- –ö–æ–≥–¥–∞ cap(slice) == len(slice)
–ï—Å–ª–∏ len(slice) —É–∂–µ —Ä–∞–≤–Ω–æ cap(slice), –Ω–æ–≤—ã–π —ç–ª–µ–º–µ–Ω—Ç –Ω–µ –ø–æ–º–µ—â–∞–µ—Ç—Å—è, –ø–æ—ç—Ç–æ–º—É Go:
  1. –°–æ–∑–¥–∞—ë—Ç –Ω–æ–≤—ã–π –º–∞—Å—Å–∏–≤ —Å —É–≤–µ–ª–∏—á–µ–Ω–Ω–æ–π cap.
  2. –ö–æ–ø–∏—Ä—É–µ—Ç –≤—Å–µ —Å—Ç–∞—Ä—ã–µ —ç–ª–µ–º–µ–Ω—Ç—ã –≤ –Ω–æ–≤—ã–π –º–∞—Å—Å–∏–≤.
  3. –î–æ–±–∞–≤–ª—è–µ—Ç –Ω–æ–≤—ã–π —ç–ª–µ–º–µ–Ω—Ç –≤ –∫–æ–Ω–µ—Ü.
  4. –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Å—Å—ã–ª–∫—É –Ω–∞ –Ω–æ–≤—ã–π –º–∞—Å—Å–∏–≤.

**–ö–æ–≥–¥–∞ —Å–æ–∑–¥–∞–µ—Ç—Å—è –Ω–æ–≤—ã–π –º–∞—Å—Å–∏–≤ –∏ —Å—Å—ã–ª–∫–∞ –Ω–∞ –Ω–µ–≥–æ –ø–µ—Ä–µ–¥–∞–µ—Ç—Å—è –≤ –ø–µ—Ä–µ–º–µ–Ω–Ω—É—é, —Å—Ç–∞—Ä—ã–π –º–∞—Å—Å–∏–≤ –æ—Å—Ç–∞–µ—Ç—Å—è –≤ –ø–∞–º—è—Ç–∏, –ø–æ–∫–∞ –Ω–∞ –Ω–µ–≥–æ –µ—Å—Ç—å —Å—Å—ã–ª–∫–∏.**

**–ö–∞–∫ –≤—ã—á–∏—Å–ª—è–µ—Ç—Å—è –Ω–æ–≤—ã–π —Ä–∞–∑–º–µ—Ä –º–∞—Å—Å–∏–≤–∞ (cap)?**

–ê–ª–≥–æ—Ä–∏—Ç–º —Ä–æ—Å—Ç–∞ –µ–º–∫–æ—Å—Ç–∏ (cap) –≤ Go —Ä–∞–±–æ—Ç–∞–µ—Ç —Å–ª–µ–¥—É—é—â–∏–º –æ–±—Ä–∞–∑–æ–º:

1. –ï—Å–ª–∏ cap = 0, —Ç–æ –ø—Ä–∏ –¥–æ–±–∞–≤–ª–µ–Ω–∏–∏ –Ω–æ–≤–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞ –∏–ª–∏ —Ä—è–¥–∞ —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –±—É–¥–µ—Ç –≤—ã–¥–µ–ª–µ–Ω–æ –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ–µ –∫–æ–ª-–≤–æ –ø–∞–º—è—Ç–∏ –¥–ª—è –∑–∞–ø–∏—Å–∏ –∏ cap –±—É–¥–µ—Ç —Ä–∞–≤–µ–Ω len
2. –ï—Å–ª–∏ cap < 1024, —Ç–æ –Ω–æ–≤–∞—è cap = 2 * —Å—Ç–∞—Ä–∞—è cap (—É–¥–≤–æ–µ–Ω–∏–µ).
3. –ï—Å–ª–∏ cap ‚â• 1024, —Ç–æ –Ω–æ–≤–∞—è cap —É–≤–µ–ª–∏—á–∏–≤–∞–µ—Ç—Å—è –ø—Ä–∏–º–µ—Ä–Ω–æ –Ω–∞ 25% (–Ω–µ —É–¥–≤–∞–∏–≤–∞–µ—Ç—Å—è, —á—Ç–æ–±—ã –Ω–µ —Ä–∞—Å—Ö–æ–¥–æ–≤–∞—Ç—å —Å–ª–∏—à–∫–æ–º –º–Ω–æ–≥–æ –ø–∞–º—è—Ç–∏).

**–¢–∞–∫–∂–µ –ø—Ä–∏ –≤—ã–¥–µ–ª–µ–Ω–∏–∏ –ø–∞–º—è—Ç–∏ –ø—Ä–∏–º–µ–Ω—è–µ—Ç—Å—è –º–µ—Ö–∞–Ω–∏–∑–º –≤—ã—Ä–∞–≤–Ω–∏–≤–∞–Ω–∏—è –ø–∞–º—è—Ç–∏, –∫–æ—Ç–æ—Ä—ã–π –º–æ–∂–µ—Ç –∏–∑–º–µ–Ω–∏—Ç—å –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –≤—ã–¥–µ–ª–µ–Ω–Ω—ã—Ö —è—á–µ–µ–∫ –ø–∞–º—è—Ç–∏:**

```go
    var slice1 []int32 // len = 0, cap = 0
    slice1 = append(slice1, 4, 5, 6) // len = 3, cap = 3
    slice1 = append(slice1, 4)
    fmt.Println(len(slice1), cap(slice1)) // len = 4, cap = 4   
    
    var slice2 []int64 // len = 0, cap = 0
    slice2 = append(slice2, 4, 5, 6) // len = 3, cap = 3
    slice2 = append(slice2, 4)
    fmt.Println(len(slice2), cap(slice2)) // len = 4, cap = 6
```

## Map

`map` –≤ Go ‚Äî —ç—Ç–æ –≤—Å—Ç—Ä–æ–µ–Ω–Ω—ã–π –∞—Å—Å–æ—Ü–∏–∞—Ç–∏–≤–Ω—ã–π –∫–æ–Ω—Ç–µ–π–Ω–µ—Ä (—Å–ª–æ–≤–∞—Ä—å), –∫–æ—Ç–æ—Ä—ã–π —Ö—Ä–∞–Ω–∏—Ç –ø–∞—Ä—ã –∫–ª—é—á‚Äì–∑–Ω–∞—á–µ–Ω–∏–µ.  
–ö–ª—é—á–∏ –¥–æ–ª–∂–Ω—ã –±—ã—Ç—å —Å—Ä–∞–≤–Ω–∏–º—ã–º–∏ (comparable) —Ç–∏–ø–∞–º–∏, —Ç.–µ. –ø–æ–¥–¥–µ—Ä–∂–∏–≤–∞—Ç—å –æ–ø–µ—Ä–∞—Ç–æ—Ä `==` –∏ `!=` (–Ω–∞–ø—Ä–∏–º–µ—Ä: `string`, `int`, `bool`, `struct` –±–µ–∑ —Å—Ä–µ–∑–æ–≤ –∏ map –≤–Ω—É—Ç—Ä–∏).  
–ó–Ω–∞—á–µ–Ω–∏—è –º–æ–≥—É—Ç –±—ã—Ç—å –ª—é–±—ã–º–∏ —Ç–∏–ø–∞–º–∏, –≤–∫–ª—é—á–∞—è —Å—Ä–µ–∑—ã, —Å—Ç—Ä—É–∫—Ç—É—Ä—ã, –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å—ã –∏ –¥–∞–∂–µ –¥—Ä—É–≥–∏–µ –∫–∞—Ä—Ç—ã.

**–û—Å–æ–±–µ–Ω–Ω–æ—Å—Ç–∏:**

- **–•–µ—à-—Ç–∞–±–ª–∏—Ü–∞ –ø–æ–¥ –∫–∞–ø–æ—Ç–æ–º** ‚Äî –∫–ª—é—á–∏ —Ö—ç—à–∏—Ä—É—é—Ç—Å—è, –∏ —Ö–µ—à –∏—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è –¥–ª—è –ø–æ–∏—Å–∫–∞ –∑–Ω–∞—á–µ–Ω–∏—è.
- **–ê–º–æ—Ä—Ç–∏–∑–∏—Ä–æ–≤–∞–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å O(1)** –¥–ª—è –æ–ø–µ—Ä–∞—Ü–∏–π –¥–æ–±–∞–≤–ª–µ–Ω–∏—è, –ø–æ–ª—É—á–µ–Ω–∏—è –∏ —É–¥–∞–ª–µ–Ω–∏—è.
- **–ù–µ—É–ø–æ—Ä—è–¥–æ—á–µ–Ω–Ω–æ—Å—Ç—å** ‚Äî –ø–æ—Ä—è–¥–æ–∫ –æ–±—Ö–æ–¥–∞ —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –Ω–µ –≥–∞—Ä–∞–Ω—Ç–∏—Ä–æ–≤–∞–Ω –∏ –º–æ–∂–µ—Ç –º–µ–Ω—è—Ç—å—Å—è –ø—Ä–∏ –∫–∞–∂–¥–æ–π –∏—Ç–µ—Ä–∞—Ü–∏–∏.
- **–ù—É–ª–µ–≤–∞—è –∏–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è** ‚Äî –ø–µ—Ä–µ–º–µ–Ω–Ω–∞—è —Ç–∏–ø–∞ `map` –ø–æ —É–º–æ–ª—á–∞–Ω–∏—é –∏–º–µ–µ—Ç –∑–Ω–∞—á–µ–Ω–∏–µ `nil`, –∏ —Å –Ω–µ–π –Ω–µ–ª—å–∑—è —Ä–∞–±–æ—Ç–∞—Ç—å –±–µ–∑ –∏–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏–∏.
- **–ê–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ —É–≤–µ–ª–∏—á–µ–Ω–∏–µ —Ä–∞–∑–º–µ—Ä–∞** ‚Äî –ø—Ä–∏ –∑–∞–ø–æ–ª–Ω–µ–Ω–∏–∏ –≤–Ω—É—Ç—Ä–µ–Ω–Ω–∏–µ –±–∞–∫–µ—Ç—ã –ø–µ—Ä–µ—Å–æ–∑–¥–∞—é—Ç—Å—è –∏ –ø–µ—Ä–µ—Ä–∞—Å–ø—Ä–µ–¥–µ–ª—è—é—Ç—Å—è.

**–ö–∞–∫ —Ä–∞–±–æ—Ç–∞–µ—Ç –ø–æ–¥ –∫–∞–ø–æ—Ç–æ–º:**


- Go —Ö—Ä–∞–Ω–∏—Ç –∫–∞—Ä—Ç—É –∫–∞–∫ **–º–∞—Å—Å–∏–≤–∞ –±–∞–∫–µ—Ç–æ–≤** (buckets), –∫–∞–∂–¥—ã–π –±–∞–∫–µ—Ç ‚Äî –Ω–∞–±–æ—Ä –∑–∞–ø–∏—Å–µ–π (—á–∞—Å—Ç—å —Ö–µ—à–∞ + –∫–ª—é—á + –∑–Ω–∞—á–µ–Ω–∏–µ).
- –î–ª—è –ø–æ–∏—Å–∫–∞ –∑–Ω–∞—á–µ–Ω–∏—è:

`–¢–µ–≥–∏ (struct tags)` ‚Äî —ç—Ç–æ —Å—Ç—Ä–æ–∫–æ–≤—ã–µ –º–µ—Ç–∫–∏, –∫–æ—Ç–æ—Ä—ã–µ –º–æ–∂–Ω–æ –ø—Ä–∏–≤—è–∑–∞—Ç—å –∫ –∫–∞–∂–¥–æ–º—É –ø–æ–ª—é —Å—Ç—Ä—É–∫—Ç—É—Ä—ã.

### –û–±–æ—Ä–∞—á–∏–≤–∞–Ω–∏–µ

### –†–∞–±–æ—Ç–∞ —Å –ø–∞–Ω–∏–∫–æ–π (panic)

`–ü–∞–Ω–∏–∫–∞ (panic)` ‚Äî —ç—Ç–æ –∞–≤–∞—Ä–∏–π–Ω–æ–µ –∑–∞–≤–µ—Ä—à–µ–Ω–∏–µ —Ç–µ–∫—É—â–µ–π –≥–æ—Ä—É—Ç–∏–Ω—ã —Å —Ä–∞–∑–≤–æ—Ä–∞—á–∏–≤–∞–Ω–∏–µ–º —Å—Ç–µ–∫–∞ –∏ –≤—ã–ø–æ–ª–Ω–µ–Ω–∏–µ–º defer. –ï—ë –º–æ–∂–Ω–æ –ø–µ—Ä–µ—Ö–≤–∞—Ç–∏—Ç—å —Ç–æ–ª—å–∫–æ —Å –ø–æ–º–æ—â—å—é recover –≤–Ω—É—Ç—Ä–∏ –æ—Ç–ª–æ–∂–µ–Ω–Ω–æ–π —Ñ—É–Ω–∫—Ü–∏–∏. –ü—Ä–∏–º–µ–Ω—è–µ—Ç—Å—è –¥–ª—è –ø—Ä–æ–≥—Ä–∞–º–º–Ω—ã—Ö –æ—à–∏–±–æ–∫, –Ω–∞—Ä—É—à–µ–Ω–∏–π –∏–Ω–≤–∞—Ä–∏–∞–Ω—Ç–æ–≤ –∏ ¬´–Ω–µ–≤–æ–∑–º–æ–∂–Ω—ã—Ö¬ª —Å–æ—Å—Ç–æ—è–Ω–∏–π ‚Äî –Ω–µ –¥–ª—è –æ–∂–∏–¥–∞–µ–º—ã—Ö –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å—Å–∫–∏—Ö –æ—à–∏–±–æ–∫ (—Ç–∞–º –Ω—É–∂–µ–Ω error)

–û—Å–æ–±–µ–Ω–Ω–æ—Å—Ç–∏:

- panic ‚Üí —Ä–∞–∑–º–æ—Ç–∫–∞ —Å—Ç–µ–∫–∞ + defer. –í—Å–µ –æ—Ç–ª–æ–∂–µ–Ω–Ω—ã–µ —Ñ—É–Ω–∫—Ü–∏–∏ –≤—ã–ø–æ–ª–Ω—è—Ç—Å—è (LIFO), –∑–∞—Ç–µ–º –ø–∞–Ω–∏–∫–∞ –ø–æ–π–¥—ë—Ç –≤—ã—à–µ –ø–æ —Å—Ç–µ–∫—É.

- –ì—Ä–∞–Ω–∏—á–Ω—ã–µ –∑–æ–Ω—ã (boundaries). –ò–¥–∏–æ–º–∞—Ç–∏—á–Ω–æ –ø–µ—Ä–µ—Ö–≤–∞—Ç—ã–≤–∞—Ç—å –ø–∞–Ω–∏–∫—É –Ω–∞ –≥—Ä–∞–Ω–∏—Ü–µ –≥–æ—Ä—É—Ç–∏–Ω—ã, main –∏–ª–∏ –≤ HTTP‚Äëmiddleware, –ª–æ–≥–∏—Ä–æ–≤–∞—Ç—å –∏ –∫–æ–Ω–≤–µ—Ä—Ç–∏—Ä–æ–≤–∞—Ç—å –≤ 500/error.
